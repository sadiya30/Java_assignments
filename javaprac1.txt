public class Math
{

  public static void main (String[]args)
  {
      int a=29;
      int b=3;
      System.out.print ("The numbers used for operation are :" +a);
      System.out.println(",\t"+b);
      int sum=a+b;
      System.out.println ("The sum of both the number is : "+sum);
      int difference=a-b;
      System.out.println ("The difference of both the number is : "+difference);
      int product=a*b;
     System.out.println ("The multiplication of both the number is : "+product);
     float division=(float)a/b;
     System.out.println ("The division of both the number is : "+division);
  }
}
